public function getFileUrl($fileName, $localPath = 'backEnd/image/teammember')
    {
        if (empty($fileName)) {
            return $localPath == 'backEnd/image/teammember/profilepic'
                ? asset('backEnd/image/dummy-user.png')
                : '#';
        }

        // agar temp path aa raha hai to usko ignore karo
        if (
            str_contains($fileName, 'xampp/tmp') ||
            str_contains($fileName, 'xampp\\tmp') ||
            str_contains($fileName, '.tmp')
        ) {
            return '#';
        }

        $fileName = str_replace('\\', '/', trim($fileName));

        $localFile = public_path($localPath . '/' . basename($fileName));

        if (file_exists($localFile)) {
            return asset($localPath . '/' . basename($fileName));
        }

        try {
            if (Storage::disk('s3')->exists($fileName)) {
                return Storage::disk('s3')->temporaryUrl($fileName, now()->addMinutes(30));
            }

            $s3Path = 'candidateonboarding/' . basename($fileName);

            if (Storage::disk('s3')->exists($s3Path)) {
                return Storage::disk('s3')->temporaryUrl($s3Path, now()->addMinutes(30));
            }
        } catch (\Exception $e) {
            \Log::error('MinIO file error: ' . $e->getMessage());
        }

        return $localPath == 'backEnd/image/teammember/profilepic'
            ? asset('backEnd/image/teammember/profilepic/16092382843.png')
            : '#';
    }

public function assignmentfileDownload($id)
{

        ->where('id', $id)

 public function allteamsubmitted()
  {
    $get_date = DB::table('timesheetreport as t')
      ->leftJoin('teammembers as tm', 'tm.id', '=', 't.teamid')
      ->leftJoin('teammembers as p', 'p.id', '=', 't.partnerid')
      ->select(
        DB::raw('MIN(t.id) as id'),
        't.teamid',
        't.week',
        DB::raw('SUM(t.totaldays) as totaldays'),
        DB::raw('SUM(t.totaltime) as totaltime'),
        DB::raw('SUM(t.dayscount) as dayscount'),
        DB::raw('MIN(t.startdate) as startdate'),
        DB::raw('MAX(t.enddate) as enddate'),
        DB::raw('MAX(t.created_at) as created_at'),
        'tm.team_member',
        'tm.emailid',
        DB::raw('MAX(p.team_member) as partnername'),
        DB::raw('MAX(t.partnerid) as partnerid')
      )
      ->groupBy('t.teamid', 't.week', 'tm.team_member', 'tm.emailid')
      ->orderByDesc(DB::raw('MAX(t.created_at)'))
      ->limit(50)
      ->get();

    $get_date->each(function ($item) {
      $item->staffcode = $this->getStaffCodeAt($item->teamid, $item->created_at);
    });

    $permissiontimesheet = DB::table('timesheetreport')->first();

    return view('backEnd.timesheet.myteamindex', compact('get_date', 'permissiontimesheet'));
  }

 public function handle($request, Closure $next)
    {
        $user = auth()->user();
        // Check if user is authenticated and inactive
        if (auth()->check() && $user->status == 0) {

            auth()->logout();
            return redirect()->route('login')->withErrors('Your account has been deactivated.');
        }

        if (auth()->check() && $user->two_factor_code && !in_array($user->email, ['Poojakumari@capitall.io', 'manojsangwan@capitall.io', 'sukhbahadur1993@gmail.com', 'sangwanm242@gmail.com', 'Guneet@capitall.io', 'Harish@capitall.io', 'Simmi@capitall.io', 'shahidraza@capitall.io'])) {
            if ($user->two_factor_expires_at->lt(now())) {
                $user->resetTwoFactorCode();
                auth()->logout();

                return redirect()->route('login')
                    ->withMessage('The two factor code has expired. Please login again.');
            }

            if (!$request->is('verify*')) {
                return redirect()->route('verify.index');
            }
        }

        return $next($request);
    }
